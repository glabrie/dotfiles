Services running on NixOS hosts.

* Forgejo

#+begin_src nix :tangle forgejo.nix
{ ... }:
{
  flake.modules.nixos.forgejo =
  { ... }:
  {
    services.forgejo = {
      enable = true;
      domain = "git.ghil.dev";
      settings = {
        server = {
          HTTP_PORT = 3000;
          ROOT_URL = "https://git.ghil.dev/";
          DOMAIN = "git.ghil.dev";
        };
        service = {
          DISABLE_REGISTRATION = true;
        };
      };
    };
  };
}
#+end_src

* Nginx

#+begin_src nix :tangle nginx.nix
{ ... }:
{
  flake.modules.nixos.nginx =
  { ... }:
  {
    services.nginx = {
      enable = true;
      virtualHosts."git.ghil.dev" = {
        enableACME = true;
        forceSSL = true;
        locations."/" = {
          proxyPass = "http://localhost:3000";
          proxyWebsockets = true;
        };
      };
    };
  };
}
#+end_src
